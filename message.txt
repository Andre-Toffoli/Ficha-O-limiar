<script>
// BARRAS
function updateBar(id, value){
    document.getElementById(id).style.width = value + "%";
}

// VALORES
function getVal(id){ return parseInt(document.getElementById(id).value); }
function setVal(id,val){ document.getElementById(id).value = val; }

// CRAFT ITEM
function craftItem(nome,item1,item2){
    if(getVal(item1)>=1 && getVal(item2)>=1){
        setVal(item1,getVal(item1)-1);
        setVal(item2,getVal(item2)-1);
        document.getElementById("equipamentos").value += nome + "\n";
        document.getElementById("craftMsg").innerText = nome + " criado e adicionado!";
    } else {
        document.getElementById("craftMsg").innerText = "Recursos insuficientes!";
    }
}

// PREVIEW IMAGEM
document.getElementById("foto").addEventListener("change", function(event){
    const reader = new FileReader();
    reader.onload = function(){
        const img = document.getElementById("preview");
        img.src = reader.result;
        img.style.display = "block";
    }
    reader.readAsDataURL(event.target.files[0]);
});

// EXPORTAR WORD
function exportarWord() {
    var conteudo = document.getElementById("conteudoExportar").innerHTML;
    var blob = new Blob(['\ufeff', conteudo], { type: 'application/msword' });
    var link = document.createElement("a");
    link.href = URL.createObjectURL(blob);
    link.download = "Ficha_O_Limiar.doc";
    link.click();
}

// MELHORIA ARMAS TLOU
function melhorarArma(tipo){
    let estrelaId = tipo === "curta" ? "estrelaCurta" : "estrelaLonga";
    let estrelas = parseInt(document.getElementById(estrelaId).innerText);

    if(estrelas >= 5){
        document.getElementById("craftMsg").innerText = "Máximo de melhorias atingido!";
        return;
    }

    let custoPecas = (estrelas + 1) * 10;
    let custoLaminas = (estrelas + 1) * 2;

    if(getVal("pecas") >= custoPecas && getVal("laminas") >= custoLaminas){
        setVal("pecas", getVal("pecas") - custoPecas);
        setVal("laminas", getVal("laminas") - custoLaminas);
        document.getElementById(estrelaId).innerText = estrelas + 1;
        document.getElementById("craftMsg").innerText = "Arma melhorada para ⭐ " + (estrelas+1);
    } else {
        document.getElementById("craftMsg").innerText =
        "Necessário: " + custoPecas + " peças e " + custoLaminas + " lâminas.";
    }
}

// REMOVER MELHORIA
function removerMelhoria(tipo){
    let estrelaId = tipo === "curta" ? "estrelaCurta" : "estrelaLonga";
    let estrelas = parseInt(document.getElementById(estrelaId).innerText);

    if(estrelas <= 0){
        document.getElementById("craftMsg").innerText = "A arma já está no nível mínimo.";
        return;
    }

    let custoPecas = estrelas * 10;
    let custoLaminas = estrelas * 2;

    setVal("pecas", getVal("pecas") + Math.floor(custoPecas/2));
    setVal("laminas", getVal("laminas") + Math.floor(custoLaminas/2));

    document.getElementById(estrelaId).innerText = estrelas - 1;
    document.getElementById("craftMsg").innerText = "Melhoria removida.";
}

// ATIRAR
function atirar(idMunicao){
    let municao = getVal(idMunicao);
    if(municao > 0){
        setVal(idMunicao, municao - 1);
        document.getElementById("craftMsg").innerText = "Tiro efetuado!";
    } else {
        document.getElementById("craftMsg").innerText = "Sem munição!";
    }
}

// CRAFT MELHORIA
function craftMelhoria(){
    if(getVal("pecas") >= 5 && getVal("laminas") >= 1){
        document.getElementById("craftMsg").innerText = "Você possui recursos para melhorar sua arma.";
    } else {
        document.getElementById("craftMsg").innerText = "Recursos insuficientes para melhoria.";
    }
}

// SAVE / LOAD
let ficha = {};
function salvarFicha() {
    ficha = {
        nome: document.getElementById("nome").value,
        idade: getVal("idade"),
        cidade: document.getElementById("cidade").value,
        estado: document.getElementById("estado").value,
        pais: document.getElementById("pais").value,
        vida: getVal("vida"),
        esforco: getVal("esforco"),
        sanidade: getVal("sanidade"),
        infeccao: getVal("infeccao"),
        esquiva: getVal("esquiva"),
        defesa: getVal("defesa"),
        fome: getVal("fome"),
        agua: getVal("agua"),
        armaCurta: document.getElementById("armaCurta").value,
        estrelaCurta: getVal("estrelaCurta"),
        municaoCurta: getVal("municaoCurta"),
        armaLonga: document.getElementById("armaLonga").value,
        estrelaLonga: getVal("estrelaLonga"),
        municaoLonga: getVal("municaoLonga"),
        pecas: getVal("pecas"),
        laminas: getVal("laminas")
    };
    localStorage.setItem("saveJogo", JSON.stringify(ficha));
    alert("Jogo salvo com sucesso!");
}

function carregarFicha() {
    let save = localStorage.getItem("saveJogo");
    if(save){
        ficha = JSON.parse(save);
        document.getElementById("nome").value = ficha.nome;
        setVal("idade", ficha.idade);
        document.getElementById("cidade").value = ficha.cidade;
        document.getElementById("estado").value = ficha.estado;
        document.getElementById("pais").value = ficha.pais;
        setVal("vida", ficha.vida); updateBar("vidaBar", ficha.vida);
        setVal("esforco", ficha.esforco); updateBar("esforcoBar", ficha.esforco);
        setVal("sanidade", ficha.sanidade); updateBar("sanidadeBar", ficha.sanidade);
        setVal("infeccao", ficha.infeccao); updateBar("infeccaoBar", ficha.infeccao);
        setVal("esquiva", ficha.esquiva); updateBar("esquivaBar", ficha.esquiva);
        setVal("defesa", ficha.defesa); updateBar("defesaBar", ficha.defesa);
        setVal("fome", ficha.fome); updateBar("fomeBar", ficha.fome);
        setVal("agua", ficha.agua); updateBar("aguaBar", ficha.agua);
        document.getElementById("armaCurta").value = ficha.armaCurta;
        setVal("estrelaCurta", ficha.estrelaCurta);
        setVal("municaoCurta", ficha.municaoCurta);
        document.getElementById("armaLonga").value = ficha.armaLonga;
        setVal("estrelaLonga", ficha.estrelaLonga);
        setVal("municaoLonga", ficha.municaoLonga);
        setVal("pecas", ficha.pecas);
        setVal("laminas", ficha.laminas);
        alert("Jogo carregado com sucesso!");
    } else {
        alert("Nenhum save encontrado!");
    }
}

function apagarSave() {
    localStorage.removeItem("saveJogo");
    alert("Save apagado com sucesso!");
}
